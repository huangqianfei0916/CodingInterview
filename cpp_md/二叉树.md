## 判断一颗二叉树是不是平衡二叉树
```cpp
class Solution {
public:
    bool isBalanced(TreeNode* root) {
        bool flag = true;
        if (root == nullptr) {
            return true;
        }
        int d = judge(root);
        if (d == -1) {
            flag = false;
        }
        return flag;

    }
    int judge(TreeNode* root) {
        if (root == nullptr) {
            return 0;
        }
        int left = judge(root->left);
        if (left == -1) {
            return -1;
        }
        int right = judge(root->right);
        if (right == -1) {
            return -1;
        }

        return abs(left - right) > 1 ? -1 : max(left, right) + 1;
    }

};
```
## 判断一颗二叉树是不是对称二叉树
```cpp
class Solution {
public:
    bool isSymmetric(TreeNode* root) {
        if (root == nullptr) {
            return true;
        }
        bool flag = judge_dc(root->left, root->right);
        return flag;
    }

    bool judge_dc(TreeNode* p1, TreeNode* p2) {
        if (p1 == nullptr && p2 == nullptr) {
            return true;
        }
        if (p1 == nullptr || p2 == nullptr) {
            return false;
        }
        if (p1->val != p2->val) {
            return false;
        }
        return judge_dc(p1->left, p2->right) && judge_dc(p1->right, p2->left);
    }
};
```
## 输入一颗二叉树，输出它的镜像
```cpp
class Solution {
public:
    TreeNode* mirrorTree(TreeNode* root) {
        if (root == nullptr) {
            return root;
        }

        TreeNode* temp = root->left;
        root->left = root->right;
        root->right = temp;
        mirrorTree(root->left);
        mirrorTree(root->right);
        return root;
    }
};
```
## 二叉树的前中序遍历

class Solution {
public:
    vector<int> preorderTraversal(TreeNode* root) {
        vector<int> vec;
        if (root == nullptr) {
            return vec;
        }
        stack<TreeNode*> s;
        TreeNode* p = root;
        while (!s.empty() || p != nullptr) {
            if (p!= nullptr) {
                s.push(p);
                vec.push_back(p->val);
                p = p->left;
            } else {
                TreeNode* temp =  s.top();
                s.pop();
                p = temp->right;
            }
        }
        return vec;
    }
};

class Solution {
public:
    vector<int> postorderTraversal(TreeNode* root) {
        vector<int> vec;
        if (root == nullptr) {
            return vec;
        }
        stack<TreeNode*> s;
        s.push(root);
        deque<int> q;
        while (!s.empty()) {
            TreeNode* temp = s.top();
            s.pop();
            q.push_front(temp->val);
            if (temp->left) {
                s.push(temp->left);
            }
            if (temp->right) {
                s.push(temp->right);
            }
        }
        for(auto it = q.begin(); it != q.end(); ++it){
            vec.push_back(*it);
        }
        return vec;
    }
};
